        function calculateAdmissions() {

            const gpa = parseFloat(document.getElementById('gpa').value) || 0;

            const sat = parseInt(document.getElementById('sat').value) || 0;

            const location = document.getElementById('location').value;

            const schoolType = document.getElementById('schoolType').value;

            

            if (gpa === 0) {

                alert('Please enter your GPA to continue');

                return;

            }

            

            let colleges = [];

            if (schoolType === 'all') {

                colleges = [...californiaColleges.uc, ...californiaColleges.csu, ...californiaColleges.private];

            } else if (schoolType === 'uc') {

                colleges = californiaColleges.uc;

            } else if (schoolType === 'csu') {

                colleges = californiaColleges.csu;

            } else if (schoolType === 'private') {

                colleges = californiaColleges.private;

            }

            

            // Filter by location

            if (location !== 'anywhere') {

                colleges = colleges.filter(college => college.location === location);

            }

            

            // Calculate admission probability

            colleges.forEach(college => {

                let probability = 50; // Base probability

                

                // GPA factor

                if (gpa >= college.avgGPA + 0.2) probability += 30;

                else if (gpa >= college.avgGPA) probability += 15;

                else if (gpa >= college.avgGPA - 0.2) probability -= 10;

                else probability -= 25;

                

                // SAT factor (if provided)

                if (sat > 0) {

                    if (sat >= college.avgSAT + 50) probability += 20;

                    else if (sat >= college.avgSAT) probability += 10;

                    else if (sat >= college.avgSAT - 50) probability -= 5;

                    else probability -= 15;

                }

                

                // Acceptance rate factor

                probability += (college.acceptance - 30);

                

                college.probability = Math.max(5, Math.min(95, probability));

            });

            

            displayResults(colleges);

        }

        

        function findColleges() {

            calculateAdmissions();

        }

        

        function displayResults(colleges) {

            const resultsDiv = document.getElementById('results');

            const matchesDiv = document.getElementById('collegeMatches');

            

            // Sort by probability

            colleges.sort((a, b) => b.probability - a.probability);

            

            let html = '';

            colleges.forEach(college => {

                let category = 'safety';

                if (college.probability < 30) category = 'reach';

                else if (college.probability < 60) category = 'target';

                

                html += `

                    <div class="college-match ${category}">

                        <div>

                            <strong>${college.name}</strong><br>

                            <small>Avg GPA: ${college.avgGPA} | Avg SAT: ${college.avgSAT} | Acceptance: ${college.acceptance}%</small>

                        </div>

                        <div class="probability">${college.probability}%</div>

                    </div>

                `;

            });

            

            matchesDiv.innerHTML = html;
